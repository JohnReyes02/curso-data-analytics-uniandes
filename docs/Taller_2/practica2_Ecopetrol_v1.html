# Sintaxis para crear 5 grupos con cortes pesonalizados

def categorizar(x):
    if x <= 155:
        return 'GRUPO1'
    elif x <= 168:
        return 'GRUPO2'
    elif x <= 195:
        return 'GRUPO3'
    else:
        return 'GRUPO4'
